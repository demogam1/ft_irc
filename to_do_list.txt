MODE CHANNEL:

o : removing operator
{
    [MODE] {CHANNEL} {COMMANDE} {TARGET NICK}
       1      2            3           4 

    REMOVE OPERATOR ROLE OF A CHANNEL FROM MY OWN CHANNEL
    input cmd --> mode #terre -o tata
    response from  server -->  :tata!~a@62.129.8.171 MODE #terre -o tata

    GIVING OPERATOR ROLE OF A CHANNEL WHICH HIS NOT MINE
    input cmd --> mode #terre +o tata
    response from server -->  :tantalum.libera.chat 482 tata #terre :You're not a channel operator

    GIVING OPERATOR ROLE TO SOMEONE WHO ISNT IN THE CHANNEL
    input --> mode #blabla +o test
    response from server --> :sodium.libera.chat 441 tata test #blabla :They aren't on that channel
}

i : giving invite
{
    [MODE] {CHANNEL} {COMMANDE} {ignored}
    
    SETTING THE CHANNEL ON INVITED MODE ONLY
    input --> mode #blabla +i
    response from server --> :tata!~a@62.129.8.171 MODE #blabla +i 

    REMOVING INVITED MODE ONLY ON THE CHANNEL
    input --> mode #blabla -i
    response from server --> :tata!~a@62.129.8.171 MODE #blabla -i
}

t :setting topic in channel
{
    [MODE] {CHANNEL} {COMMANDE} {ignored}

    input --> mode #ref -t
    response from server --> :tata!~a@62.129.8.171 MODE #ref -t

    input --> mode #ref +t
    response from server --> :tata!~a@62.129.8.171 MODE #ref +t
}
